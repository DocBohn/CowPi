108#define cowpi_initialize_i2c do { \
109 /* Set SCL Frequency [100kHz] = CPU Clock Frequency [16MHz] / (16 + 2 * TWBR * prescaler [1]) */ \
110 TWBR = 72; \
111 TWSR &= ~((1 << TWPS1) | (1 << TWPS0)); \
112 /* We won't rely on interrupts since students will work with display before knowing interrupts. */ \
113 /* We're also not going to enable acknowledgements: the basic labs will only use \
114 * controller-transmitter mode, and without interrupts we can't honor the ACK anyway. */ \
115 /* So we'll just enable TWI */ \
116 TWCR = (1 << TWEN) | (1 << TWEA); \
117} while(0)
